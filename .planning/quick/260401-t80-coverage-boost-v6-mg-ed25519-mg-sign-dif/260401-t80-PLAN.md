With threshold-based check:
```python
def check_coverage():
    section("CODE COVERAGE -- Genetor Analysis")
    THRESHOLD = 49.0  # Current coverage ~40%, set threshold just below to avoid immediate fail
    out, code = run("python scripts/agent_coverage.py --summary")
    if code != 0:
        warn(f"Coverage tool failed -- {out.strip()}")
        return True  # tool failure is advisory, not a hard fail
    # Parse coverage percentage from output
    import re
    match = re.search(r'(\d+\.?\d*)%', out)
    if match:
        pct = float(match.group(1))
        if pct >= THRESHOLD:
            ok(f"Coverage {pct:.1f}% >= {THRESHOLD}% threshold")
            return True
        else:
            err(f"Coverage {pct:.1f}% < {THRESHOLD}% threshold")
            return False
    else:
        ok(f"Coverage analysis complete -- {out.strip()}")
        return True
```
